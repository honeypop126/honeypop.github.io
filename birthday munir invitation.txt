    <script>
        let envelopeOpened = false;
        
        function openEnvelope() {
            if (!envelopeOpened) {
                document.getElementById('envelope').classList.add('opened');
                envelopeOpened = true;
            }
        }
        
        function handleYes(event) {
            event.stopPropagation();
            document.getElementById('homepage').style.display = 'none';
            document.getElementById('datePage').style.display = 'block';
        }
        
        function handleNo(event) {
            event.stopPropagation();
            document.getElementById('modal').style.display = 'flex';
        }
        
        function closeModal() {
            document.getElementById('modal').style.display = 'none';
        }
        
        function selectDate(dateType) {
            const messages = {
                picnic: "Perfect! I'll pack the strawberries and lemonade! 🍓",
                cottage: "Wonderful! I'll light the candles and pick fresh rosemary! 🕯️",
                market: "Amazing! I know the best flower stall! 🌻",
                stargazing: "Magical choice! I'll bring the softest blankets! ⭐"
            };
            
            // Create a custom modal for the date selection
            const modal = document.getElementById('modal');
            const modalText = modal.querySelector('.modal-text');
            modalText.textContent = messages[dateType];
            modal.style.display = 'flex';
        }
        
        // Add some sparkle effects when hovering over date options
        document.querySelectorAll('.date-option').forEach(option => {
            option.addEventListener('mouseenter', function() {
                this.style.background = 'linear-gradient(135deg, rgba(255,255,255,0.95), rgba(248,244,240,0.95))';
            });
            
            option.addEventListener('mouseleave', function() {
                this.style.background = 'rgba(255,255,255,0.8)';
            });
        });
        
        // Close modal when clicking outside
        document.getElementById('modal').addEventListener('click', function(e) {
            if (e.target === this) {
                closeModal();
            }
        });
    </script>
